def load_and_cache_segments(engine, cache_dir: str = "segments_cache", chunk_size: int = 50000) -> pd.DataFrame:
    """
    Loads segments, expands timbre and pitch into separate columns,
    caches chunks as Parquet, and returns concatenated DataFrame.
    """

    # Define column names for timbre and pitch.
    timbre_cols = [f"timbre_{i+1}" for i in range(12)]
    pitch_cols = [f"pitch_{i+1}" for i in range(12)]

    # Check if cache directory exists and has Parquet files.
    if os.path.isdir(cache_dir) and any(file.endswith(".parquet") for file in os.listdir(cache_dir)):

        # Load all of the parquet files from the segments cache directory in batches.
        parts = sorted(part for part in os.listdir(cache_dir) if part.endswith(".parquet"))

        # Set a batch size for loading the cached files.
        batch_size = 10
        dfs = []

        # Process the cached files in batches to avoid memory issues.
        for i in range(0, len(parts), batch_size):

            # Get the current batch of parquet files according to the batch size.
            batch_parts = parts[i:i + batch_size]

            # Load the current batch and concatenate them into a single DataFrame.
            batch_df = pd.concat([pd.read_parquet(os.path.join(cache_dir, part)) for part in batch_parts], ignore_index=True)

            # Append the batch DataFrame to the list of DataFrames.
            dfs.append(batch_df)

            # Print progress.
            print(f"Loaded cached batch {i//batch_size + 1}/{(len(parts)-1)//batch_size + 1}")

        # Concatenate all batches into a single DataFrame and return.
        return pd.concat(dfs, ignore_index=True)

    # Create cache directory if it doesn't exist.
    os.makedirs(cache_dir, exist_ok=True)
    
    # Define SQL statement to extract segments with related track and genre info.
    statement = text("""
    SELECT
      s.track_id AS track_pk,
      (elem ->> 'segment_index')::int AS segment_index,
      (elem -> 'timbre') AS timbre,
      (elem -> 'pitch') AS pitch,
      (elem ->> 'loudness_max')::float AS loudness_max,
      (elem ->> 'confidence')::float AS confidence,
      g.genre AS genre
    FROM segment AS s
    JOIN track AS t ON s.track_id = t.id
    JOIN genre AS g ON t.genre_id = g.id
    CROSS JOIN LATERAL jsonb_array_elements(s.segments) AS elem
    ORDER BY s.track_id, segment_index;
    """)

    # Process and cache segments in chunks.
    with engine.connect() as conn:

        # Read and process segments in chunks.
        for idx, chunk in enumerate(pd.read_sql_query(statement, conn, chunksize=chunk_size)):
            # Expand timbre and pitch columns
            chunk[timbre_cols] = pd.DataFrame(chunk["timbre"].tolist(), index=chunk.index)
            chunk[pitch_cols] = pd.DataFrame(chunk["pitch"].tolist(), index=chunk.index)

            # Drop original array columns
            chunk.drop(["timbre", "pitch"], axis=1, inplace=True)

            # Save chunk to Parquet
            filename = os.path.join(cache_dir, f"seg_cache_{idx:04d}.parquet")
            chunk.to_parquet(filename, index=False)
            print(f"Wrote expanded chunk {idx + 1} ({len(chunk)} rows) → {filename}")

    # Reload cached chunks
    parts = sorted(part for part in os.listdir(cache_dir) if part.endswith(".parquet"))
    return pd.concat([pd.read_parquet(os.path.join(cache_dir, part)) for part in parts], ignore_index=True)

def prepare_sequences(df, sequence_length, feature_columns):
    sequences = []
    valid_track_ids = []  # Store valid track_pk values

    for track_id in df['track_pk'].unique():
        track_data = df[df['track_pk'] == track_id].sort_values('segment_index')
        features = track_data[feature_columns].values

        if len(features) < sequence_length:
            padding = np.zeros((sequence_length - len(features), len(feature_columns)))
            features = np.vstack([features, padding])
        else:
            features = features[:sequence_length]

        sequences.append(features)
        valid_track_ids.append(track_id)

    return np.array(sequences), valid_track_ids

def prepare_datasets_with_ids(X, ids, test_size, val_size, seed=42):
    """
    Prepare train, validation, and test datasets (with aligned track ids).
    """
    # Create train and test sets (with ids)
    X_train, X_test, ids_train, ids_test = train_test_split(X, ids, test_size=test_size, random_state=seed, shuffle=True)

    # Create train and validation sets (with ids)
    X_train, X_val, ids_train, ids_val = train_test_split(X_train, ids_train, test_size=val_size, random_state=seed, shuffle=True)

    return X_train, X_val, X_test, ids_train, ids_val, ids_test

def build_model(sequence_length, feature_dim):
    """
    Build the CRNN model.
    """
    inputs = Input(shape=(sequence_length, feature_dim))

    # 1st conv layer
    x = Conv1D(128, kernel_size=3, padding='same', activation='elu', kernel_regularizer=l2(0.007))(inputs)
    x = MaxPooling1D(pool_size=2)(x)
    x = BatchNormalization()(x)
    x = Dropout(0.6)(x)

    # 2nd conv layer
    x = Conv1D(64, kernel_size=3, padding='same', activation='elu', kernel_regularizer=l2(0.007))(x)
    x = MaxPooling1D(pool_size=2)(x)
    x = BatchNormalization()(x)
    x = Dropout(0.6)(x)

    # 3rd conv layer
    x = Conv1D(32, kernel_size=3, padding='same', activation='elu', kernel_regularizer=l2(0.007))(x)
    x = MaxPooling1D(pool_size=2)(x)
    x = BatchNormalization()(x)
    x = Dropout(0.6)(x)

    # GRU layers
    x = GRU(32, return_sequences=True)(x)
    x = GRU(32)(x)

    # Embedding branch (128-dim)
    embedding_output = Dense(128, name='embedding')(x)

    # Projection head (for contrastive loss during training only)
    p = Dense(256, activation='elu', kernel_regularizer=l2(0.007))(embedding_output)
    p = Dropout(0.3)(p)
    projection_output = Dense(128, name='projection')(p)


    # Two views of the same backbone:
    train_model = Model(inputs, projection_output, name="crnn_supcon_train")  # train on 'projection'
    embed_model = Model(inputs, embedding_output,  name="crnn_embed")         # use 'embedding' at inference
    return train_model, embed_model

def supcon_loss(y_true, z, temperature=0.1):
    """
    Supervised contrastive loss with track ids as y_true.
    Pulls together embeddings with the same track_id, pushes others apart.
    """
    y_true = tf.reshape(tf.cast(y_true, tf.int32), [-1, 1])                 # [B,1]
    mask   = tf.cast(tf.equal(y_true, tf.transpose(y_true)), tf.float32)    # [B,B]

    z = tf.math.l2_normalize(z, axis=1)                                     # cosine sims
    logits = tf.matmul(z, z, transpose_b=True) / temperature                # [B,B]
    logits -= tf.reduce_max(logits, axis=1, keepdims=True)                  # stability

    # remove self-comparisons
    logits_mask = 1.0 - tf.eye(tf.shape(logits)[0])
    mask *= logits_mask

    exp_logits = tf.exp(logits) * logits_mask
    log_prob = logits - tf.math.log(tf.reduce_sum(exp_logits, axis=1, keepdims=True) + 1e-12)

    pos_cnt = tf.reduce_sum(mask, axis=1)
    mean_log_prob_pos = tf.reduce_sum(mask * log_prob, axis=1) / (pos_cnt + 1e-12)

    return -tf.reduce_mean(mean_log_prob_pos)

def make_random_crops(X, ids, crop_len=150, n_views=4, rng_seed=42):
    """
    From each (T,F) sequence, make n_views random crops of length crop_len.
    Returns stacked crops (N*n_views, crop_len, F) and repeated track ids.
    """
    rng = np.random.default_rng(rng_seed)
    N, T, F = X.shape
    X_out, y_out = [], []
    for i in range(N):
        for _ in range(n_views):
            start = 0 if T <= crop_len else rng.integers(0, T - crop_len + 1)
            X_out.append(X[i, start:start+crop_len, :])
            y_out.append(ids[i])
    return np.asarray(X_out), np.asarray(y_out)

class SupConSequence(Sequence):
    """
     
    """
    def __init__(self, X, ids, n_pos=4, tracks_per_batch=32, seed=42):
        self.X = X
        self.ids = np.asarray(ids)
        self.n_pos = n_pos
        self.tracks_per_batch = tracks_per_batch
        self.rng = np.random.default_rng(seed)

        # map track_id -> list of indices with that id
        self.id_to_idx = {}
        for i, track_id in enumerate(self.ids):
            self.id_to_idx.setdefault(track_id, []).append(i)
        self.unique_ids = np.array(list(self.id_to_idx.keys()))

        # determine number of steps per epoch
        self.steps = int(np.ceil(len(self.unique_ids) / self.tracks_per_batch))
        self.on_epoch_end()

    def __len__(self):
        return self.steps

    def on_epoch_end(self):
        # shuffle track ids and their indices
        self.rng.shuffle(self.unique_ids)
        for track_id in self.unique_ids:
            idxs = self.id_to_idx[track_id]
            self.rng.shuffle(idxs)

    def __getitem__(self, i):
        # get track ids for this batch
        start = i * self.tracks_per_batch
        end   = min((i+1) * self.tracks_per_batch, len(self.unique_ids))
        batch_ids = self.unique_ids[start:end]

        batch_x, batch_y = [], []
        for track_id in batch_ids:
            idxs = self.id_to_idx[track_id]
            # if not enough views, sample with replacement
            if len(idxs) >= self.n_pos:
                pick = idxs[:self.n_pos]
            else:
                pick = list(idxs) + list(self.rng.choice(idxs, size=self.n_pos - len(idxs)))
            for p in pick:
                batch_x.append(self.X[p])
                batch_y.append(self.ids[p])

        return np.asarray(batch_x), np.asarray(batch_y, dtype='int32')

def l2n(A): 
    # L2-normalize rows of A
    return A / (np.linalg.norm(A, axis=1, keepdims=True) + 1e-9)

def recall_at_k_loocv(E_q, E_db, ids_q, ids_db, K=1):
    hits = 0
    for i in range(len(E_q)):
        sims = E_db @ E_q[i]
        sims[i] = -1e9                 # exclude the query itself
        topk = np.argsort(-sims)[:K]
        hits += np.any(ids_db[topk] == ids_q[i])
    return hits / len(E_q)

def plot_tsne(X_emb, y_ids, sample_size=2000, max_tracks=10, seed=42):
    rng = np.random.default_rng(seed)
    uniq, counts = np.unique(y_ids, return_counts=True)
    keep = uniq[np.argsort(-counts)[:max_tracks]]
    mask = np.isin(y_ids, keep)
    X, y = X_emb[mask], y_ids[mask]
    if len(X) > sample_size:
        idx = rng.choice(len(X), size=sample_size, replace=False)
        X, y = X[idx], y[idx]
    perp = max(5, min(30, len(X)//3))
    Z = TSNE(n_components=2, perplexity=perp, init='pca', learning_rate='auto',
             random_state=seed).fit_transform(X)
    plt.figure(figsize=(6,6))
    for tid in np.unique(y):
        m = y==tid
        plt.scatter(Z[m,0], Z[m,1], s=10, alpha=0.8, label=f"track {tid}")
    plt.title(f"t-SNE (perplexity={perp})"); plt.xlabel("t-SNE-1"); plt.ylabel("t-SNE-2")
    plt.legend(markerscale=2); plt.grid(True); plt.show()
